#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <map>

using namespace std;
int n, m;
char arr[11][11]; // n: 1 - n-2까지 / m: 1 - m-2 까지, 8888 까지 존재
int hx, hy; // 구멍
int rx, ry; // red
int bx, by; // blue
int dx[4] = {-1, 0, 1, 0};
int dy[4] = {0, 1, 0, -1};

map<string, int> visited;

string toIndex(int rx, int ry, int bx, int by){
    string res = "";
    res += to_string(rx);
    res += to_string(ry);
    res += to_string(bx);
    res += to_string(by);
    return res;
}

pair<int, int> slide(int dire, int x, int y, char c){
    while(true){
        int nx = x + dx[dire];
        int ny = y + dy[dire];

        if(arr[nx][ny] == 'R' || arr[nx][ny] == 'B' || arr[nx][ny] == '#'){
            break;
        }else if(arr[nx][ny] == 'O'){

            x = nx;
            y = ny;
            // 업데이트 해 주고나서 반환(O인 위치로!)
            break;
        }
        x = nx;
        y = ny;
    }

    if(arr[x][y] != 'O'){
        arr[x][y] = c;
    }

    return {x, y};
}

string move(int dire, int rx, int ry, int bx, int by) {
    // 굴러가는 방향
    pair<int, int> r = {rx, ry};
    pair<int, int> b = {bx, by};
    if(dire == 0){
        // 위방향
        if(ry == by){
            // 한줄에 있기때문에 x가 작은것부터 움직인다
            if(rx < bx){
                r = slide(dire, rx, ry, 'R');
                b = slide(dire, bx, by, 'B');
            }else{
                b = slide(dire, bx, by, 'B');
                r = slide(dire, rx, ry, 'R');
            }
        }else{
            // 순서 상관없다
            b = slide(dire, bx, by, 'B');
            r = slide(dire, rx, ry, 'R');
        }
    }else if(dire == 1){
        // 오른쪽 방향
        if(rx == bx){
            if(ry < by){
                b = slide(dire, bx, by, 'B');
                r = slide(dire, rx, ry, 'R');
            }else{
                r = slide(dire, rx, ry, 'R');
                b = slide(dire, bx, by, 'B');
            }
        }else{
            // 순서 상관없다
            b = slide(dire, bx, by, 'B');
            r = slide(dire, rx, ry, 'R');
        }
    }
    else if(dire == 2){
        if(ry == by){
          // 한줄에 있으니까 x가 큰것 부터 움직인다
          if(rx < bx){
              b = slide(dire, bx, by, 'B');
              r = slide(dire, rx, ry, 'R');
          }else{
              r = slide(dire, rx, ry, 'R');
              b = slide(dire, bx, by, 'B');
          }
        }else{
            // 순서 상관없다
            b = slide(dire, bx, by, 'B');
            r = slide(dire, rx, ry, 'R');
        }
    }else if(dire == 3){
        if(rx == bx){
            // 한줄에 있으니까 y가 작은것 부터 움직인다
            if(ry < by){
                r = slide(dire, rx, ry, 'R');
                b = slide(dire, bx, by, 'B');
            }else{
                b = slide(dire, bx, by, 'B');
                r = slide(dire, rx, ry, 'R');
            }
        }else{
            // 순서 상관없다
            b = slide(dire, bx, by, 'B');
            r = slide(dire, rx, ry, 'R');
        }
    }


    return toIndex(r.first, r.second, b.first, b.second);
}
